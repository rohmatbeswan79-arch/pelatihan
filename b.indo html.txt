        <div id="wordContainer"></div>

        <textarea id="answerArea" placeholder="Tarik kata ke sini..." ondrop="drop(event)" ondragover="allowDrop(event)"></textarea>

<script>
    const correctAnswer = "saya makan nasi di samping toko";
    let words = ["saya", "makan", "nasi", "di", "samping", "toko"];

    // Fungsi untuk mengacak kata
    function shuffle(array) {
        return array.sort(() => Math.random() - 0.5);
    }

    // Menampilkan kata secara acak
    function displayWords() {
        const container = document.getElementById("wordContainer");
        container.innerHTML = "";
        shuffle(words).forEach(word => {
            const span = document.createElement("span");
            span.textContent = word;
            span.className = "word-box";
            span.draggable = true;
            span.ondragstart = drag;
            container.appendChild(span);
        });
    }

    function allowDrop(event) {
        event.preventDefault();
    }

    function drag(event) {
        event.dataTransfer.setData("text", event.target.textContent);
        event.target.style.opacity = "0.5";
    }

    function drop(event) {
        event.preventDefault();
        const word = event.dataTransfer.getData("text");
        const textarea = document.getElementById("answerArea");
        textarea.value += (textarea.value ? " " : "") + word;

        // Hapus kata yang sudah digunakan
        const boxes = document.querySelectorAll(".word-box");
        boxes.forEach(box => {
            if (box.textContent === word && box.style.opacity === "0.5") {
                box.remove();
            }
        });
    }

    function checkAnswer() {
        const userAnswer = document.getElementById("answerArea").value.trim();
        const feedback = document.getElementById("feedback");
        if (userAnswer === correctAnswer) {
            feedback.textContent = "Bagus kawan! âœ…";
        } else {
            feedback.textContent = "Maaf jawaban anda salah, hadehhh ulangi lagi ðŸ˜…";
        }
    }

    function resetGame() {
        document.getElementById("answerArea").value = "";
        document.getElementById("feedback").textContent = "";
        displayWords();
    }

    // Menampilkan kata pertama kali
    displayWords();
</script>
